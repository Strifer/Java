1   package hr.fer.zemris.java.simplecomp.impl.instructions;
2   
3   import java.util.List;
4   
5   import hr.fer.zemris.java.simplecomp.models.Computer;
6   import hr.fer.zemris.java.simplecomp.models.Instruction;
7   import hr.fer.zemris.java.simplecomp.models.InstructionArgument;
8   /**
9    * Razred predstavlja naredbu load koju naše računalo može izvoditi.
10   * <br><code>liad r0, LABELA</code>
11   * <br>U r0 spremamo sadržaj spremljen na memorijskoj lokaciji LABELA
12   * @author Filip Džidić
13   *
14   */
15  public class InstrLoad implements Instruction {
16  	/** registar u kojeg spremamo vrijednost iz memorije */
17  	private int registerIndex;
18  	/** memorijska lokacija na kojoj je sačuvan podatak koji čitamo */
19  	private int location;
20  	
21  	/**
22  	 * Konstruktor koji inicijalizira vrijednosti registara za daljnje računanje.
23  	 * Prvi parametar mora biti registar opće manije a drugi argument je memorijska lokacija s koje čitamo.
24  	 * @param arguments lista parametera koji je parser pročitao za našu naredbu
25  	 */
26  	public InstrLoad(List<InstructionArgument> arguments) {
27  		if (arguments.size()!=2) {
28  			throw new IllegalArgumentException("Expected 2 arguments!");
29  		}
30  		if (!arguments.get(0).isRegister()) {
31  			throw new IllegalArgumentException("Type mismatch for argument 0!");
32  		}
33  		if (!arguments.get(1).isNumber()) {
34  			throw new IllegalArgumentException("Type mismatch for argument 1");
35  		}
36  		this.registerIndex=((Integer) arguments.get(0).getValue()).intValue();
37  		this.location=((Integer) arguments.get(1).getValue()).intValue();
38  	}
39  	
40  	@Override
41  	public boolean execute(Computer computer) {
42  		Object value = computer.getMemory().getLocation(location);
43  		computer.getRegisters().setRegisterValue(registerIndex, value);
44  		return false;
45  	}
46  
47  }
