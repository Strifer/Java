3   import hr.fer.zemris.java.simplecomp.models.Computer;
4   import hr.fer.zemris.java.simplecomp.models.ExecutionUnit;
5   import hr.fer.zemris.java.simplecomp.models.Instruction;
6   /**
7    * Ovaj razred predstavlja upravljačku jedinicu našeg računalnog sustava.
8    * Vrši dohvat, dekodiranje i izvođenje naredbi sve dok se ne dogodi greška ili dok se ne izvede naredba halt
9    * @author Filip Džidić
10   *
11   */
12  public class ExecutionUnitImpl implements ExecutionUnit {
13  
14  	@Override
15  	/**
16  	 * Metoda koja pokreće izvedbu našeg računala vraća false ako izađemo iz nje zbog neke greške.
17  	 */
18  	public boolean go(Computer computer) {
19  		computer.getRegisters().setProgramCounter(0);
20  		boolean haltExecuted = false;
21  		while (!haltExecuted) {
22  			int programCount = computer.getRegisters().getProgramCounter();
23  			Instruction instruction = (Instruction) computer.getMemory().getLocation(programCount);
24  			computer.getRegisters().incrementProgramCounter();
25  			try {
26  				haltExecuted = instruction.execute(computer);
27  			} catch (IllegalArgumentException ex) {
28  				return false;
29  			}
30  			
31  		}
32  		return true;
33  	}
34  
35  }
