1   package hr.fer.zemris.java.simplecomp.impl.instructions;
2   
3   import java.util.List;
4   import hr.fer.zemris.java.simplecomp.models.Computer;
5   import hr.fer.zemris.java.simplecomp.models.Instruction;
6   import hr.fer.zemris.java.simplecomp.models.InstructionArgument;
7   /**
8    * Razred predstavlja naredbu mul koju naše računalo može izvoditi.
9    * <br><code>mul r0, r1, r2</code>
10   * <br>Rezultat r1*r2 spremamo u r0
11   * @author Filip Džidić
12   *
13   */
14  public class InstrMul implements Instruction {
15  	/** prvi operand naredbe */
16  	private int indexRegister1;
17  	/** drugi operand naredbe */
18  	private int indexRegister2;
19  	/** rezultantni operator naredbe */
20  	private int indexRegister3;
21  	
22  	/**
23  	 * Konstruktor koji inicijalizira vrijednosti registara za daljnje računanje.
24  	 * Svi parametri moraju biti registri.
25  	 * @param arguments lista parametera koji je parser pročitao za našu naredbu
26  	 */
27  	public InstrMul(List<InstructionArgument> arguments) {
28  		if (arguments.size() != 3) {
29  			throw new IllegalArgumentException("Expected 3 arguments!");
30  		}
31  		if (!arguments.get(0).isRegister()) {
32  			throw new IllegalArgumentException("Type mismatch for argument 0!");
33  		}
34  		if (!arguments.get(1).isRegister()) {
35  			throw new IllegalArgumentException("Type mismatch for argument 1!");
36  		}
37  		if (!arguments.get(2).isRegister()) {
38  			throw new IllegalArgumentException("Type mismatch for argument 2!");
39  		}
40  		this.indexRegister1 = ((Integer) arguments.get(0).getValue())
41  				.intValue();
42  		this.indexRegister2 = ((Integer) arguments.get(1).getValue())
43  				.intValue();
44  		this.indexRegister3 = ((Integer) arguments.get(2).getValue())
45  				.intValue();
46  	}
47  
48  	public boolean execute(Computer computer) {
49  		Object value1 = computer.getRegisters()
50  				.getRegisterValue(indexRegister2);
51  		Object value2 = computer.getRegisters()
52  				.getRegisterValue(indexRegister3);
53  		computer.getRegisters().setRegisterValue(
54  				indexRegister1,
55  				Integer.valueOf(((Integer) value1).intValue()
56  						* ((Integer) value2).intValue()));
57  		return false;
58  	}
59  }
