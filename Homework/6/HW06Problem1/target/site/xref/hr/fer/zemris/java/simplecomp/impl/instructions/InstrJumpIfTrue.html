5   import hr.fer.zemris.java.simplecomp.models.Computer;
6   import hr.fer.zemris.java.simplecomp.models.Instruction;

9   /**
10   * Razred predstavlja naredbu jumpIfTrue koju naše računalo može izvoditi. <br>
11   * <code>jumpIfTrue LABELA</code> <br>
12   * Programsko brojilo preusmjeravamo na lokaciju labele i izvodimo naredbe na
13   * sljedećim memorijskim lokacijama ali samo ako je uvjet ispunjen (postavljena
14   * zastavica)
15   * 
16   * @author Filip Džidić
17   *
18   */
19  public class InstrJumpIfTrue implements Instruction {
20  	/** lokacija na koju preusmjeravamo programsko brojilo */
21  	private int location;
22  	
23  	/**
24  	 * Konstruktor koji inicijalizira naredbu i provjerava ispravnost argumenata.
25  	 * Naredba mora imati element memorijske lokaciju da bi znao gdje skočiti pri izvedbi.
26  	 * @param arguments argumenti uz naredbu
27  	 */
28  	public InstrJumpIfTrue(List<InstructionArgument> arguments) {
29  		if (arguments.size() != 1) {
30  			throw new IllegalArgumentException("Expected 1 arguments!");
31  		}
32  		if (!arguments.get(0).isNumber()) {
33  			throw new IllegalArgumentException("Type mismatch for argument 0!");
34  		}
35  
36  		this.location = ((Integer) arguments.get(0).getValue()).intValue();
37  	}
38  
39  	@Override
40  	public boolean execute(Computer computer) {
41  		if (computer.getRegisters().getFlag()) {
42  			computer.getRegisters().setProgramCounter(location);
43  		}
44  		return false;
45  	}
46  
47  }
